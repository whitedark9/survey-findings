import data from "~/constants/data/data";
import findings from "~/constants/data/findings";
import { Callout } from "nextra/components";
import SimpleBarChart from "~/ui/simple-bar-chart";
import PieChartWithLabel from "~/ui/pie-chart-with-label";
import clsx from "clsx";

{Object.keys(data).map((key, index) => {

let i = 0;

return (

  <div key={key} className="pb-12 border-b-1 mb-12">

<h2 className="mb-3 text-xl font-bold">
  {index + 1}. {key.toUpperCase().replaceAll("_", " ")}
</h2>

{
Object.keys(data[key]).map((table_title, idx)=> {

    if (Array.isArray(data[key][table_title])) {
      i++;

      return (
        <div key={table_title} className="mb-70 p-3 pb-20 border-b-1 last:border-b-0">
          <h3 className="text-lg mb-3 font-bold capitalize">table {index+1}.{i} {table_title.replaceAll("_", " ")}</h3>

          <div className={"flex flex-col lg:flex-row items-start py-6 gap-4 lg:gap-0"}>

<SimpleBarChart data={data[key][table_title]} />

<PieChartWithLabel data={data[key][table_title]} />

</div>

          <Callout type={"info"}>
            FINDINGS: {findings[table_title]}
          </Callout>
        </div>
      )

    }

})
}

</div>
)
})}
